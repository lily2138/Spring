def myrange(x):
    if(x%5==0):
        fl=x-5
        ci=x
        return "("+str(fl)+","+str(ci)+"]"
    ci=(x//5+1)*5
    fl=x//5*5
    return "("+str(fl)+","+str(ci)+"]"
